CREATE TABLE movies ( movieID int UNIQUE NOT NULL AUTO_INCREMENT, title varchar(256) NOT NULL, location varchar(4096) NOT NULL, genres varchar (256) DEFAULT '', playcount int(10) DEFAULT '0', actors varchar(256) DEFAULT '', popularity FLOAT(10,7) UNSIGNED DEFAULT '0.0', releasedate DATE, director varchar(256) DEFAULT '', length TIME DEFAULT '00:00:00', descrition TEXT);
CREATE TABLE tvshows ( showID int UNIQUE NOT NULL AUTO_INCREMENT, title varchar(256) NOT NULL, location varchar(4096) NOT NULL, genres varchar (256) DEFAULT '', playcount int(10) DEFAULT '0', actors varchar(256) DEFAULT '', popularity FLOAT(10,7) UNSIGNED DEFAULT '0.0', releasedate DATE, director varchar(256) DEFAULT '', length TIME DEFAULT '00:00:00', season int(2) DEFAULT '0', episode int(3) DEFAULT '0', descrition TEXT);
CREATE TABLE music ( songID int UNIQUE NOT NULL AUTO_INCREMENT, title varchar(256) NOT NULL, location varchar(4096) NOT NULL, genres varchar (256) DEFAULT '', playcount int(10) DEFAULT '0', artist varchar(256) DEFAULT '', featuredartists varchar(256) DEFAULT '', album varchar(256) DEFAULT '', length TIME DEFAULT '00:00:00', releasedate DATE);
